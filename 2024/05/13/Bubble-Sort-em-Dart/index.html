Bubble Sort em Dart

O Bubble Sort é um algoritmo de ordenação simples, mas ineficiente, que percorre repetidamente a lista, compara elementos adjacentes e os troca se estiverem na ordem errada. Aqui está uma implementação em Dart:

void bubbleSort(List<int> lista) {
    int n = lista.length;
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (lista[j] > lista[j + 1]) {
                int temp = lista[j];
                lista[j] = lista[j + 1];
                lista[j + 1] = temp;
            }
        }
    }
}

void main() {
    List<int> numeros = [5, 2, 8, 1, 9];
    print('Lista original: $numeros');
    bubbleSort(numeros);
    print('Lista ordenada: $numeros');
}
